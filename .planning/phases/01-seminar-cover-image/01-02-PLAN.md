```typescript
export const uploadCoverImage = async (seminarId: string, file: File): Promise<Seminar> => {
  const formData = new FormData();
  formData.append('file', file);
  return await apiClient.post<Seminar>(
    `${BASE_URL}/${seminarId}/cover-image`,
    formData,
    { headers: { 'Content-Type': 'multipart/form-data' } }
  );
};

export const deleteCoverImage = async (seminarId: string): Promise<Seminar> => {
  return await apiClient.delete<Seminar>(`${BASE_URL}/${seminarId}/cover-image`);
};
```

```typescript
export const useUploadCoverImageMutation = () => {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: ({ seminarId, file }: { seminarId: string; file: File }) =>
      seminarService.uploadCoverImage(seminarId, file),
    onSuccess: () => {
      toast.success('Imagen de portada actualizada');
      queryClient.invalidateQueries({ queryKey: ['seminars'] });
    },
    onError: () => {
      toast.error('Error al subir la imagen');
    },
  });
};

export const useDeleteCoverImageMutation = () => {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: (seminarId: string) => seminarService.deleteCoverImage(seminarId),
    onSuccess: () => {
      toast.success('Imagen de portada eliminada');
      queryClient.invalidateQueries({ queryKey: ['seminars'] });
    },
    onError: () => {
      toast.error('Error al eliminar la imagen');
    },
  });
};
```

Note: The toast messages for upload/delete errors are NOT shown for client-side validation errors (file type, file size) — those are handled inline in the CoverImageDropZone component (plan 01-03). These toasts cover network errors only.
